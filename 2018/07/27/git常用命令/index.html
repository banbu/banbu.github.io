git常用命令简单整理

git init //初始化仓库
git status //查看工作区的状态
git branch //查看分支
git branch -r //查看远程分支
git brach -a //查看本地-远程所有分支
git checkout 分支名 //切换分支
git checkout -b 分支名 //本地创建分支，同时切换到该分支，提交的时候会再服务端创建一个分支
git pull //更新服务端代码到本地(检查是否更新完服务端代码)
git merge 分支名称 //合并分支到当前分支
git push //提交代码到服务端
git log //当前分支历史版本

git checkout . && git clean -xdf //删除新增内容

强制更新服务端代码到本地
git fetch –all
git reset –hard origin/master //替换为要拉取的远程分支名称
git pull
